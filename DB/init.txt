CREATE TABLE systems(
	id SERIAL PRIMARY KEY NOT NULL,
	sysname VARCHAR(50) NOT NULL
);
CREATE TABLE system_account(
	id SERIAL PRIMARY KEY NOT NULL,
	systems_id integer references systems,
	sysname VARCHAR(50) NOT NULL
);
CREATE TABLE users(
	id SERIAL PRIMARY KEY NOT NULL,
	system_account_id integer references system_account,
	nickname VARCHAR(50) NOT NULL
);
CREATE TABLE info_u(
	id SERIAL PRIMARY KEY NOT NULL,
	users_id integer references users,
	country VARCHAR(50) NOT NULL,
	birth_data DATE NOT NULL
);
CREATE TABLE telephone(
	id SERIAL PRIMARY KEY NOT NULL,
	telephone integer NOT NULL
);
CREATE TABLE email(
	id SERIAL PRIMARY KEY NOT NULL,
	email VARCHAR(50) NOT NULL
);
CREATE TABLE full_info(
	id SERIAL PRIMARY KEY NOT NULL,
	info_u_id integer references info_u,
	telephone_id integer references telephone,
	email_id integer references email
);

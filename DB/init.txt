CREATE TABLE IF NOT EXISTS systems(
	id SERIAL PRIMARY KEY NOT NULL,
	sysname VARCHAR(50) NOT NULL
);
CREATE TABLE IF NOT EXISTS system_account(
	id SERIAL PRIMARY KEY NOT NULL,
	systems_id integer references systems,
	sysname VARCHAR(50) NOT NULL
);
CREATE TABLE IF NOT EXISTS users(
	id SERIAL PRIMARY KEY NOT NULL,
	system_account_id integer references system_account,
	nickname VARCHAR(50) NOT NULL
);
CREATE TABLE IF NOT EXISTS info_u(
	id SERIAL PRIMARY KEY NOT NULL,
	users_id integer references users,
	country VARCHAR(50) NOT NULL,
	birth_data DATE NOT NULL
);
CREATE TABLE IF NOT EXISTS telephone(
	id SERIAL PRIMARY KEY NOT NULL,
	telephone integer NOT NULL
);
CREATE TABLE IF NOT EXISTS email(
	id SERIAL PRIMARY KEY NOT NULL,
	email VARCHAR(50) NOT NULL
);
CREATE TABLE IF NOT EXISTS full_info(
	id SERIAL PRIMARY KEY NOT NULL,
	info_u_id integer references info_u,
	telephone_id integer references telephone,
	email_id integer references email
);
CREATE TABLE IF NOT EXISTS game(
	id SERIAL PRIMARY KEY NOT NULL,
	gname VARCHAR(100) NOT NULL,
	gdeveloper VARCHAR(100) NOT NULL,
	grdate DATE NOT NULL,
	gprise integer NOT NULL
);
CREATE TABLE IF NOT EXISTS modification(
	id SERIAL PRIMARY KEY NOT NULL,
	mname VARCHAR(100) NOT NULL,
	mdeveloper VARCHAR(100) NOT NULL,
	mrdate DATE NOT NULL,
	mprise integer NOT NULL
);
CREATE TABLE IF NOT EXISTS electronic_wallet(
	id SERIAL PRIMARY KEY NOT NULL,
	users_id integer references users,
	balance integer NOT NULL
);
CREATE TABLE IF NOT EXISTS game_info(
	id SERIAL PRIMARY KEY NOT NULL,
	game_id integer references game,
	modification_id integer references modification
);
CREATE TABLE IF NOT EXISTS user_library(
	id SERIAL PRIMARY KEY NOT NULL,
	users_id integer references users,
	game_info_id integer references game_info
);
